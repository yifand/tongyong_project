# PDI智能监测平台数据库设计文档

## 1. 数据库选型与配置

### 1.1 版本推荐

| 配置项 | 推荐值 | 说明 |
|--------|--------|------|
| PostgreSQL版本 | 15.x 或 16.x | 支持原生分区、并行查询、JIT编译 |
| 扩展插件 | timescaledb, pg_stat_statements | 时序数据优化、查询性能分析 |

### 1.2 数据库参数配置

```sql
-- 连接配置
max_connections = 500                     -- 最大连接数
shared_buffers = 4GB                      -- 共享内存缓冲区 (建议内存的25%)
effective_cache_size = 12GB               -- 操作系统缓存估计 (建议内存的75%)
work_mem = 16MB                           -- 排序/哈希操作内存
maintenance_work_mem = 512MB              -- 维护操作内存 (VACUUM, CREATE INDEX)

-- WAL配置
wal_buffers = 16MB
max_wal_size = 4GB
min_wal_size = 1GB
wal_compression = on

-- 查询优化
random_page_cost = 1.1                    -- SSD存储设置
effective_io_concurrency = 200            -- SSD并发IO
max_parallel_workers_per_gather = 4
max_parallel_workers = 8

-- 日志配置
log_destination = 'stderr,csvlog'
logging_collector = on
log_directory = 'log'
log_filename = 'postgresql-%Y-%m-%d_%H%M%S.log'
log_rotation_age = 1d
log_rotation_size = 100MB
log_min_duration_statement = 1000         -- 记录超过1秒的查询
log_checkpoints = on
log_connections = on
log_disconnections = on
log_lock_waits = on

-- 自动清理
autovacuum = on
autovacuum_max_workers = 3
autovacuum_naptime = 10s
```

### 1.3 字符集与排序规则

```sql
-- 创建数据库
CREATE DATABASE pdi_monitor
    WITH
    ENCODING = 'UTF8'
    LC_COLLATE = 'zh_CN.UTF-8'
```

## 3. 详细字段设计

### 3.1 系统表

#### 3.1.1 sys_user (用户表)

```sql
CREATE TABLE sys_user (
    id              BIGSERIAL PRIMARY KEY COMMENT '用户ID',
    username        VARCHAR(50) NOT NULL COMMENT '用户名',
    password        VARCHAR(255) NOT NULL COMMENT '加密密码(BCrypt)',
    real_name       VARCHAR(50) COMMENT '真实姓名',
    phone           VARCHAR(20) COMMENT '手机号',
    email           VARCHAR(100) COMMENT '邮箱',
    avatar          VARCHAR(255) COMMENT '头像URL',
    site_id         BIGINT COMMENT '所属站点ID(为空表示全部站点)',
    status          SMALLINT NOT NULL DEFAULT 1 COMMENT '状态: 0-禁用, 1-启用',
    last_login_time TIMESTAMP COMMENT '最后登录时间',
    last_login_ip   VARCHAR(50) COMMENT '最后登录IP',
    created_at      TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    updated_at      TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '更新时间',
    created_by      BIGINT COMMENT '创建人ID',
    updated_by      BIGINT COMMENT '更新人ID',
    deleted         SMALLINT NOT NULL DEFAULT 0 COMMENT '删除标记: 0-未删除, 1-已删除',

    CONSTRAINT uk_sys_user_username UNIQUE (username),
    CONSTRAINT uk_sys_user_phone UNIQUE (phone)
) COMMENT '用户表';
```

#### 3.1.2 sys_role (角色表)

```sql
CREATE TABLE sys_role (
    id          BIGSERIAL PRIMARY KEY COMMENT '角色ID',
    role_code   VARCHAR(50) NOT NULL COMMENT '角色编码',
    role_name   VARCHAR(50) NOT NULL COMMENT '角色名称',
    description VARCHAR(255) COMMENT '角色描述',
    data_scope  SMALLINT DEFAULT 1 COMMENT '数据范围: 1-全部, 2-本站点, 3-本人',
    status      SMALLINT NOT NULL DEFAULT 1 COMMENT '状态: 0-禁用, 1-启用',
    sort_order  INT DEFAULT 0 COMMENT '排序',
    created_at  TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    updated_at  TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '更新时间',

    CONSTRAINT uk_sys_role_code UNIQUE (role_code)
) COMMENT '角色表';
```

#### 3.1.3 sys_user_role (用户角色关联表)

```sql
CREATE TABLE sys_user_role (
    id          BIGSERIAL PRIMARY KEY COMMENT 'ID',
    user_id     BIGINT NOT NULL COMMENT '用户ID',
    role_id     BIGINT NOT NULL COMMENT '角色ID',
    created_at  TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',

    CONSTRAINT uk_user_role UNIQUE (user_id, role_id),
    CONSTRAINT fk_sur_user_id FOREIGN KEY (user_id) REFERENCES sys_user(id),
    CONSTRAINT fk_sur_role_id FOREIGN KEY (role_id) REFERENCES sys_role(id)
) COMMENT '用户角色关联表';
```

#### 3.1.4 sys_permission (权限表)

```sql
CREATE TABLE sys_permission (
    id          BIGSERIAL PRIMARY KEY COMMENT '权限ID',
    parent_id   BIGINT DEFAULT 0 COMMENT '父权限ID(0为顶级)',
    perm_code   VARCHAR(100) NOT NULL COMMENT '权限编码',
    perm_name   VARCHAR(50) NOT NULL COMMENT '权限名称',

    path        VARCHAR(200) COMMENT '路由路径',
    component   VARCHAR(200) COMMENT '组件路径',
    icon        VARCHAR(50) COMMENT '图标',
    sort_order  INT DEFAULT 0 COMMENT '排序',
    status      SMALLINT NOT NULL DEFAULT 1 COMMENT '状态: 0-禁用, 1-启用',
    created_at  TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    updated_at  TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '更新时间',

    CONSTRAINT uk_sys_permission_code UNIQUE (perm_code)
) COMMENT '权限表';
```

#### 3.1.5 sys_role_permission (角色权限关联表)

```sql
CREATE TABLE sys_role_permission (
    id          BIGSERIAL PRIMARY KEY COMMENT 'ID',
    role_id     BIGINT NOT NULL COMMENT '角色ID',
    perm_id     BIGINT NOT NULL COMMENT '权限ID',
    created_at  TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',

    CONSTRAINT uk_role_perm UNIQUE (role_id, perm_id),
    CONSTRAINT fk_srp_role_id FOREIGN KEY (role_id) REFERENCES sys_role(id),
    CONSTRAINT fk_srp_perm_id FOREIGN KEY (perm_id) REFERENCES sys_permission(id)
) COMMENT '角色权限关联表';
```

#### 3.1.6 operation_log (操作日志表)

```sql
CREATE TABLE operation_log (
    id          BIGSERIAL COMMENT '日志ID',
    log_time    TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '日志时间',
    user_id     BIGINT COMMENT '用户ID',
    username    VARCHAR(50) COMMENT '用户名',
    operation   VARCHAR(100) NOT NULL COMMENT '操作类型',
    method      VARCHAR(200) COMMENT '请求方法',
    request_url VARCHAR(500) COMMENT '请求URL',
    request_params TEXT COMMENT '请求参数',
    response_data TEXT COMMENT '响应数据',
    ip_address  VARCHAR(50) COMMENT 'IP地址',
    user_agent  VARCHAR(500) COMMENT '浏览器UA',
    execute_time INT COMMENT '执行时长(ms)',
    status      SMALLINT COMMENT '状态: 0-失败, 1-成功',
    error_msg   TEXT COMMENT '错误信息',

    PRIMARY KEY (id, log_time)
) PARTITION BY RANGE (log_time) COMMENT '操作日志表';
```

```sql
CREATE TABLE channel (
```

### 3.3 业务表

#### 3.3.1 state_stream (状态流表)

```sql
CREATE TABLE state_stream (
    id              BIGSERIAL COMMENT '状态ID',
    event_time      TIMESTAMP NOT NULL COMMENT '事件时间',
    channel_id      BIGINT NOT NULL COMMENT '通道ID',
    box_id          BIGINT NOT NULL COMMENT '盒子ID',
    site_id         BIGINT NOT NULL COMMENT '站点ID',

    -- 原始检测结果
    door_open       SMALLINT NOT NULL COMMENT '门状态: 0-关, 1-开',
    person_present  SMALLINT NOT NULL COMMENT '人员存在: 0-无, 1-有',
    person_entering_exiting SMALLINT NOT NULL COMMENT '人员进出: 0-无, 1-进出中',

    -- 状态编码 S1-S8
    state_code      VARCHAR(5) NOT NULL COMMENT '状态编码: S1,S3,S5,S7,S8',

    -- 检测置信度
    door_confidence     DECIMAL(4,3) COMMENT '门检测置信度',
    person_confidence   DECIMAL(4,3) COMMENT '人员检测置信度',

    -- 原始帧信息
    frame_id        BIGINT COMMENT '帧ID',
    image_url       VARCHAR(500) COMMENT '截图URL',

    -- 处理状态
    processed       SMALLINT DEFAULT 0 COMMENT '是否已处理: 0-未处理, 1-已处理',

    PRIMARY KEY (id, event_time)
) PARTITION BY RANGE (event_time) COMMENT '状态流表';
```

#### 3.3.2 pdi_task (PDI作业表)

```sql
CREATE TABLE pdi_task (
    id                  BIGSERIAL PRIMARY KEY COMMENT '作业ID',
    task_no             VARCHAR(50) NOT NULL COMMENT '作业编号',
    channel_id          BIGINT NOT NULL COMMENT '通道ID',
    box_id              BIGINT NOT NULL COMMENT '盒子ID',
    site_id             BIGINT NOT NULL COMMENT '站点ID',

    -- 时间信息
    start_time          TIMESTAMP NOT NULL COMMENT '开始时间',
    end_time            TIMESTAMP COMMENT '结束时间(空表示进行中)',
    duration_seconds    INT COMMENT '作业时长(秒)',
    standard_duration   INT DEFAULT 1800 COMMENT '标准时长(秒,默认30分钟)',

    -- 状态流转记录
    enter_state_seq     TEXT COMMENT '进入状态序列JSON',
    exit_state_seq      TEXT COMMENT '离开状态序列JSON',

    -- 作业结果
    task_status         SMALLINT DEFAULT 1 COMMENT '状态: 1-进行中, 2-已完成, 3-异常中断',
    task_result         SMALLINT COMMENT '结果: 1-合格, 2-超时, 3-异常',

    -- 关联媒体
    start_image_url     VARCHAR(500) COMMENT '进入截图URL',
    end_image_url       VARCHAR(500) COMMENT '离开截图URL',
    video_url           VARCHAR(500) COMMENT '录像URL',

    -- 备注
    remark              VARCHAR(500) COMMENT '备注',

    created_at          TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    updated_at          TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '更新时间',

    CONSTRAINT uk_pdi_task_no UNIQUE (task_no),
    CONSTRAINT fk_pdi_channel_id FOREIGN KEY (channel_id) REFERENCES channel(id),
    CONSTRAINT fk_pdi_box_id FOREIGN KEY (box_id) REFERENCES box(id),
    CONSTRAINT fk_pdi_site_id FOREIGN KEY (site_id) REFERENCES site(id)
) COMMENT 'PDI作业表';
```

#### 3.3.3 alarm (报警记录表)

```sql
CREATE TABLE alarm (
    id              BIGSERIAL COMMENT '报警ID',

    alarm_level     SMALLINT DEFAULT 2 COMMENT '级别: 1-低, 2-中, 3-高, 4-紧急',

    -- 位置信息
    site_id         BIGINT NOT NULL COMMENT '站点ID',
    box_id          BIGINT NOT NULL COMMENT '盒子ID',
    channel_id      BIGINT NOT NULL COMMENT '通道ID',
    site_name       VARCHAR(100) COMMENT '站点名称(冗余)',
    channel_name    VARCHAR(100) COMMENT '通道名称(冗余)',

    -- 关联作业
    pdi_task_id     BIGINT COMMENT '关联PDI作业ID',

    -- 报警内容
    alarm_title     VARCHAR(200) NOT NULL COMMENT '报警标题',
    alarm_desc      VARCHAR(500) COMMENT '报警描述',

    -- 媒体证据
    image_url       VARCHAR(500) COMMENT '报警截图URL',
    video_url       VARCHAR(500) COMMENT '关联录像URL',

    -- 处理状态
    handle_status   SMALLINT DEFAULT 0 COMMENT '处理状态: 0-未处理, 1-已确认, 2-已处理, 3-误报',
    handle_time     TIMESTAMP COMMENT '处理时间',
    handle_user_id  BIGINT COMMENT '处理人ID',
    handle_remark   VARCHAR(500) COMMENT '处理备注',

    -- 扩展信息
    extra_data      JSONB COMMENT '扩展信息JSON',

    created_at      TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    updated_at      TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '更新时间',

    PRIMARY KEY (id, alarm_time)
) PARTITION BY RANGE (alarm_time) COMMENT '报警记录表';
```

## 5. 分区策略

### 5.1 分区策略概述

| 表名 | 分区类型 | 分区键 | 分区粒度 | 保留周期 |
|------|----------|--------|----------|----------|

### 5.3 分区维护函数

```sql
-- 创建新分区函数
CREATE OR REPLACE FUNCTION create_daily_partition(
    p_table_name TEXT,
    p_date DATE
) RETURNS TEXT
LANGUAGE plpgsql
AS $$
DECLARE
    partition_name TEXT;
    start_ts TIMESTAMP;
    end_ts TIMESTAMP;
BEGIN
    partition_name := p_table_name || '_' || TO_CHAR(p_date, 'YYYYMMDD');
    start_ts := p_date::TIMESTAMP;
    end_ts := (p_date + INTERVAL '1 day')::TIMESTAMP;

    EXECUTE format(
        'CREATE TABLE IF NOT EXISTS %I PARTITION OF %I
         FOR VALUES FROM (%L) TO (%L)',
        partition_name, p_table_name, start_ts, end_ts
    );

    RETURN partition_name;
END $$;

-- 创建月分区函数
CREATE OR REPLACE FUNCTION create_monthly_partition(
    p_table_name TEXT,
    p_date DATE
) RETURNS TEXT
LANGUAGE plpgsql
AS $$
DECLARE
    partition_name TEXT;
    start_ts TIMESTAMP;
    end_ts TIMESTAMP;
    month_start DATE;
BEGIN
    month_start := DATE_TRUNC('month', p_date);
    partition_name := p_table_name || '_' || TO_CHAR(month_start, 'YYYYMM');
    start_ts := month_start::TIMESTAMP;
    end_ts := (month_start + INTERVAL '1 month')::TIMESTAMP;

    EXECUTE format(
        'CREATE TABLE IF NOT EXISTS %I PARTITION OF %I
         FOR VALUES FROM (%L) TO (%L)',
        partition_name, p_table_name, start_ts, end_ts
    );

    RETURN partition_name;
END $$;

-- 删除旧分区函数
CREATE OR REPLACE FUNCTION drop_old_partitions(
    p_table_name TEXT,
    p_retention_days INT
) RETURNS TABLE(dropped_partition TEXT)
LANGUAGE plpgsql
AS $$
DECLARE
    cutoff_date TIMESTAMP;
    partition_rec RECORD;
BEGIN
    cutoff_date := CURRENT_TIMESTAMP - (p_retention_days || ' days')::INTERVAL;

    FOR partition_rec IN
        SELECT inhrelid::regclass::text AS partition_name
        FROM pg_inherits
        WHERE inhparent = p_table_name::regclass
          AND inhrelid::regclass::text NOT LIKE '%default'
    LOOP
        -- 检查分区上限是否早于保留日期
        IF (SELECT pg_catalog.pg_get_expr(relpartbound, inhrelid)
            FROM pg_class WHERE oid = partition_rec.partition_name::regclass)
            LIKE '%TO (''' || cutoff_date::date || '%' THEN

            EXECUTE format('DROP TABLE IF EXISTS %I', partition_rec.partition_name);
            dropped_partition := partition_rec.partition_name;
            RETURN NEXT;
        END IF;
    END LOOP;
END $$;
```

### 6.3 自动清理定时任务

```sql
-- 创建定时任务扩展(如使用pg_cron)
-- CREATE EXTENSION IF NOT EXISTS pg_cron;

-- 每日凌晨2点创建新分区并清理旧分区
CREATE OR REPLACE FUNCTION daily_partition_maintenance()
RETURNS void
LANGUAGE plpgsql
AS $$
BEGIN
    -- 创建未来7天的state_stream分区
    PERFORM create_daily_partition('state_stream', CURRENT_DATE + i)
    FROM generate_series(1, 7) AS i;

    -- 创建未来3个月的alarm分区
    PERFORM create_monthly_partition('alarm', CURRENT_DATE + (i || ' months')::INTERVAL)
    FROM generate_series(1, 3) AS i;

    -- 删除30天前的state_stream分区
    PERFORM * FROM drop_old_partitions('state_stream', 30);

    -- 删除90天前的alarm分区(先归档重要数据)
    -- INSERT INTO alarm_archive SELECT * FROM alarm WHERE alarm_time < NOW() - INTERVAL '90 days' AND alarm_level >= 3;
    PERFORM * FROM drop_old_partitions('alarm', 90);

    -- 删除180天前的操作日志分区
    PERFORM * FROM drop_old_partitions('operation_log', 180);
END $$;

-- 使用pg_cron设置定时任务(如已安装)
-- SELECT cron.schedule('daily-maintenance', '0 2 * * *', 'SELECT daily_partition_maintenance()');
```

### 9.1 状态流查询

```sql
-- 查询指定通道最近100条状态记录
SELECT
    event_time,
    state_code,
    door_open,
    person_present,
    person_entering_exiting,
    door_confidence,
    person_confidence
FROM state_stream
WHERE channel_id = 1
  AND event_time >= NOW() - INTERVAL '1 hour'
ORDER BY event_time DESC
LIMIT 100;

-- 查询状态转换序列(用于PDI进入/离开判定)
SELECT
    event_time,
    state_code,
    LAG(state_code) OVER (ORDER BY event_time) as prev_state
FROM state_stream
WHERE channel_id = 1
  AND event_time >= NOW() - INTERVAL '10 minutes'
ORDER BY event_time;

-- 统计各状态出现次数
SELECT
    state_code,
    COUNT(*) as count,
    MIN(event_time) as first_time,
    MAX(event_time) as last_time
FROM state_stream
WHERE event_time >= CURRENT_DATE
GROUP BY state_code
ORDER BY count DESC;
```

### 9.2 PDI作业查询

```sql
-- 查询今日PDI作业列表(分页)
SELECT
    t.id,
    t.task_no,
    t.channel_id,
    c.channel_name,
    s.site_name,
    t.start_time,
    t.end_time,
    t.duration_seconds,
    t.standard_duration,
    CASE
        WHEN t.task_status = 1 THEN '进行中'
        WHEN t.task_status = 2 THEN '已完成'
        ELSE '异常中断'
    END as status_text,
    CASE
        WHEN t.task_result = 1 THEN '合格'
        WHEN t.task_result = 2 THEN '超时'
        WHEN t.task_result = 3 THEN '异常'
        ELSE '-'
    END as result_text
FROM pdi_task t
JOIN channel c ON t.channel_id = c.id
JOIN site s ON t.site_id = s.id
WHERE t.start_time >= CURRENT_DATE
ORDER BY t.start_time DESC
LIMIT 20 OFFSET 0;

-- 统计今日PDI作业概况
SELECT
    COUNT(*) as total_tasks,
    COUNT(*) FILTER (WHERE task_status = 2) as completed_tasks,
    COUNT(*) FILTER (WHERE task_result = 2) as timeout_tasks,
    AVG(duration_seconds) FILTER (WHERE task_status = 2) as avg_duration,
    MAX(duration_seconds) FILTER (WHERE task_status = 2) as max_duration
FROM pdi_task
WHERE start_time >= CURRENT_DATE;

-- 查询超时作业
SELECT
    t.*,
    c.channel_name,
    s.site_name
FROM pdi_task t
JOIN channel c ON t.channel_id = c.id
JOIN site s ON t.site_id = s.id
WHERE t.task_result = 2
  AND t.start_time >= CURRENT_DATE - INTERVAL '7 days'
ORDER BY t.start_time DESC;
```

### 9.5 统计报表查询

```sql
-- PDI作业日报表
SELECT
    t.site_id,
    s.site_name,
    DATE(t.start_time) as work_date,
    COUNT(*) as total_tasks,
    COUNT(*) FILTER (WHERE t.task_result = 1) as qualified_tasks,
    COUNT(*) FILTER (WHERE t.task_result = 2) as timeout_tasks,
    ROUND(AVG(t.duration_seconds)::numeric, 2) as avg_duration,
    ROUND(MAX(t.duration_seconds)::numeric, 2) as max_duration
FROM pdi_task t
JOIN site s ON t.site_id = s.id
WHERE t.start_time >= CURRENT_DATE - INTERVAL '30 days'
GROUP BY t.site_id, s.site_name, DATE(t.start_time)
ORDER BY work_date DESC, s.site_name;

-- 站点报警月度统计
SELECT
    s.site_name,
    DATE_TRUNC('month', a.alarm_time) as month,
    COUNT(*) as total_alarms,
    COUNT(*) FILTER (WHERE a.alarm_type = 1) as pdi_timeout_alarms,
    COUNT(*) FILTER (WHERE a.alarm_type = 2) as smoking_alarms,
    COUNT(*) FILTER (WHERE a.handle_status = 0) as unhandled_alarms,
    ROUND(AVG(EXTRACT(EPOCH FROM (a.handle_time - a.alarm_time))/60)::numeric, 2) as avg_handle_minutes
FROM alarm a
JOIN site s ON a.site_id = s.id
WHERE a.alarm_time >= DATE_TRUNC('month', CURRENT_DATE - INTERVAL '6 months')
GROUP BY s.site_name, DATE_TRUNC('month', a.alarm_time)
ORDER BY month DESC, total_alarms DESC;

-- 通道工作效率排名
SELECT
    c.channel_name,
    s.site_name,
    COUNT(t.id) as task_count,
    ROUND(AVG(t.duration_seconds)::numeric, 2) as avg_duration,
    COUNT(t.id) FILTER (WHERE t.task_result = 2) as timeout_count,
    ROUND(100.0 * COUNT(t.id) FILTER (WHERE t.task_result = 2) / NULLIF(COUNT(t.id), 0), 2) as timeout_rate
FROM channel c
JOIN site s ON c.site_id = s.id
LEFT JOIN pdi_task t ON c.id = t.channel_id
    AND t.start_time >= CURRENT_DATE - INTERVAL '7 days'
WHERE c.status = 1
GROUP BY c.id, c.channel_name, s.site_name
HAVING COUNT(t.id) > 0
ORDER BY timeout_rate, avg_duration;
```

## 附录: 状态编码定义

| 状态编码 | door_open | person_present | person_entering_exiting | 状态描述 |
|----------|-----------|----------------|-------------------------|----------|
| S1 | 0 | 0 | 0 | 空闲(门关+无人) |
| S3 | 0 | 1 | 0 | 门关+有人(人员进入完成) |
| S5 | 1 | 0 | 0 | 门开+无人 |
| S7 | 1 | 1 | 0 | 门开+有人+未进出 |
| S8 | 1 | 1 | 1 | 门开+有人+进出中 |

**状态流转规则:**
- **进入判定**: S7 → S8 → S3 (门开有人→进出中→门关有人) = 开始PDI作业
- **离开判定**: S3 → S8 → S7 → S5/S1 (门关有人→进出中→门开有人→门开无人/空闲) = 结束PDI作业
