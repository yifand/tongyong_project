## 1. 模块职责简述

### 1.1 职责定义

算法数据入口模块（`pdi-algorithm-inlet`）是连接边缘计算层与业务平台层的**专属边界模块**，承担以下核心职责：

| 职责 | 说明 |
|------|------|
| **数据接收** | 接收边缘盒子推送的状态流、报警事件、心跳数据 |
| **数据校验** | 执行数据格式校验、盒子认证、权限验证 |
| **协议转换** | 将边缘盒子的私有数据协议转换为内部领域事件 |
| **数据分发** | 通过Spring Event将事件异步分发至规则引擎 |
| **去重幂等** | 基于`boxId+timestamp+channelId`实现幂等控制 |
| **流量削峰** | 快速响应并异步处理，接口响应时间目标 ≤ 50ms |

### 1.2 架构定位

引用《后端架构设计文档》第4.2节：

> `pdi-algorithm-inlet` 模块负责接收边缘盒子推送的状态流、报警事件、心跳数据；数据校验、协议转换、分发至规则引擎。

模块依赖关系：
- 依赖：`pdi-common`, `pdi-rule-engine`
- 被依赖：`pdi-start`

```java
@RequestMapping("/api/v1/inlet")

@Tag(name = "算法数据入口", description = "接收边缘盒子推送的数据")
public class AlgorithmInletController {

    private final AlgorithmInletService inletService;
    private final HeartbeatService heartbeatService;
    private final BoxAuthenticator boxAuthenticator;

    /**
     * 接收状态三元组流
     */
    @PostMapping("/state")
    @Operation(summary = "接收状态流数据")
    public ResponseEntity<InletResponse> receiveStateStream(
            @RequestHeader("X-Box-Token") String boxToken,

        inletService.processStateStream(request, authResult.getSiteId());

        return ResponseEntity.ok(InletResponse.success());
    }

                    .body(InletResponse.fail("BOX_AUTH_FAILED", "盒子认证失败"));
```

#### AlgorithmInletServiceImpl (实现)

```java
@Service
@RequiredArgsConstructor
@Slf4j
public class AlgorithmInletServiceImpl implements AlgorithmInletService {

    private final ApplicationEventPublisher eventPublisher;
    private final RequestValidator requestValidator;
    private final IdempotencyRepository idempotencyRepository;
    private final StateStreamRepository stateStreamRepository;
    private final AlarmRecordRepository alarmRecordRepository;

    @Override
    @Transactional
    public void processStateStream(StateStreamRequest request, Long siteId) {
        // 1. 生成幂等键
        String idempotencyKey = generateIdempotencyKey(
            request.getBoxId(),
            request.getChannelId(),
            request.getTimestamp()
        );

        // 2. 幂等检查
        if (isDuplicate(idempotencyKey)) {
            log.debug("重复状态流数据，已忽略: boxId={}, channelId={}, timestamp={}",
                request.getBoxId(), request.getChannelId(), request.getTimestamp());
            return;
        }

        // 3. 保存幂等记录
        saveIdempotencyRecord(idempotencyKey, "STATE_STREAM");

        // 4. 数据校验
        requestValidator.validateStateStream(request);

        // 5. 持久化原始数据
        StateStreamEntity entity = convertToEntity(request, siteId);
        stateStreamRepository.save(entity);

        // 6. 发布领域事件（异步处理）
        StateStreamEvent event = new StateStreamEvent(
            entity.getId(),
            siteId,
            request.getBoxId(),
            request.getChannelId(),
            request.getTimestamp(),
            request.getStateCode(),
            request.getState()
        );
        eventPublisher.publishEvent(event);

        log.debug("状态流数据处理完成: idempotencyKey={}", idempotencyKey);
    }

    @Override
    @Transactional
    public void processAlarmEvent(AlarmEventRequest request, Long siteId) {
        // 1. 生成幂等键
        String idempotencyKey = generateIdempotencyKey(
            request.getBoxId(),
            request.getChannelId(),
            request.getTimestamp()
        ) + ":" + request.getAlarmType();

        // 2. 幂等检查
        if (isDuplicate(idempotencyKey)) {
            log.debug("重复报警事件，已忽略: boxId={}, alarmType={}",
                request.getBoxId(), request.getAlarmType());
            return;
        }

        // 3. 保存幂等记录
        saveIdempotencyRecord(idempotencyKey, "ALARM_EVENT");

        // 4. 数据校验
        requestValidator.validateAlarmEvent(request);

        // 5. 持久化报警记录
        AlarmRecordEntity entity = convertToAlarmEntity(request, siteId);
        AlarmRecordEntity saved = alarmRecordRepository.save(entity);

        // 6. 发布领域事件
        AlarmEvent event = new AlarmEvent(
            saved.getId(),
            siteId,
            request.getBoxId(),
            request.getChannelId(),
            request.getAlarmType(),
            request.getTimestamp(),
            request.getImageUrl(),
            request.getConfidence(),
            request.getLocation()
        );
        eventPublisher.publishEvent(event);

        log.info("报警事件处理完成: alarmId={}, type={}", saved.getId(), request.getAlarmType());
    }

    private String generateIdempotencyKey(String boxId, String channelId, LocalDateTime timestamp) {
        return String.format("%s:%s:%s", boxId, channelId, timestamp.toString());
    }

    private boolean isDuplicate(String key) {
        return idempotencyRepository.existsByKeyAndCreatedAtAfter(
            key, LocalDateTime.now().minusMinutes(5)
        );
    }

    private void saveIdempotencyRecord(String key, String type) {
        IdempotencyRecord record = new IdempotencyRecord();
        record.setKey(key);
        record.setType(type);
        record.setExpireAt(LocalDateTime.now().plusMinutes(10));
        idempotencyRepository.save(record);
    }

    // ... convert methods
}
```

```java
@Service
@RequiredArgsConstructor
@Slf4j
public class HeartbeatServiceImpl implements HeartbeatService {

    private final ApplicationEventPublisher eventPublisher;
    private final EdgeBoxRepository edgeBoxRepository;

    @Override
    @Transactional
    public void processHeartbeat(HeartbeatRequest request, Long siteId) {
        // 1. 更新盒子状态
        EdgeBox box = edgeBoxRepository.findByBoxId(request.getBoxId())
            .orElseThrow(() -> new BoxNotFoundException(request.getBoxId()));

        box.setLastHeartbeatAt(LocalDateTime.now());
        box.setCpuUsage(request.getCpuUsage());
        box.setMemoryUsage(request.getMemoryUsage());
        box.setDiskUsage(request.getDiskUsage());
        box.setVersion(request.getVersion());
        box.setStatus(1); // 在线

        edgeBoxRepository.save(box);

        // 2. 发布心跳事件（用于监控告警）
        HeartbeatEvent event = new HeartbeatEvent(
            box.getId(),
            siteId,
            request.getBoxId(),
            request.getTimestamp(),
            request.getCpuUsage(),
            request.getMemoryUsage(),
            request.getDiskUsage()
        );
        eventPublisher.publishEvent(event);

        log.debug("心跳处理完成: boxId={}", request.getBoxId());
    }
}
```

### 2.6 认证与校验组件

#### BoxAuthenticator

```java
@Component
@RequiredArgsConstructor
public class BoxAuthenticator {

    private final EdgeBoxRepository edgeBoxRepository;
    private final CaffeineCache<String, BoxAuthResult> authCache;

    /**
     * 盒子认证
     * @param token 请求Token
     * @param boxId 盒子ID
     * @return 认证结果
     */
    public BoxAuthResult authenticate(String token, String boxId) {
        // 缓存检查
        String cacheKey = token + ":" + boxId;
        BoxAuthResult cached = authCache.getIfPresent(cacheKey);
        if (cached != null) {
            return cached;
        }

        // 数据库验证
        Optional<EdgeBox> boxOpt = edgeBoxRepository.findByBoxId(boxId);
        if (boxOpt.isEmpty()) {
            return BoxAuthResult.invalid("盒子不存在");
        }

        EdgeBox box = boxOpt.get();

        // Token验证 (使用HMAC-SHA256)
        String expectedToken = generateToken(box);
        if (!expectedToken.equals(token)) {
            return BoxAuthResult.invalid("Token无效");
        }

        BoxAuthResult result = BoxAuthResult.valid(box.getSiteId(), box.getId());
        authCache.put(cacheKey, result);
        return result;
    }

    private String generateToken(EdgeBox box) {
        // 使用boxSecret + boxId + 日期生成Token
        String data = box.getBoxId() + ":" + LocalDate.now().toString();
        return HmacUtils.hmacSha256Hex(box.getBoxSecret(), data);
    }
}
```

#### RequestValidator

```java
@Component
@RequiredArgsConstructor
public class RequestValidator {

    private final ChannelRepository channelRepository;

    /**
     * 校验状态流请求
     */
    public void validateStateStream(StateStreamRequest request) {
        // 校验状态码与三元组一致性
        int expectedCode = calculateStateCode(request.getState());
        if (!request.getStateCode().equals(expectedCode)) {
            throw new InvalidStateException(
                String.format("状态码不一致: 期望%d, 实际%d", expectedCode, request.getStateCode())
            );
        }

        // 校验通道存在性
        validateChannelExists(request.getBoxId(), request.getChannelId());

        // 校验时间戳合理性 (不能是未来时间，不能是过去太久)
        validateTimestamp(request.getTimestamp());
    }

    /**
     * 校验报警事件请求
     */
    public void validateAlarmEvent(AlarmEventRequest request) {
        // 校验通道存在性
        validateChannelExists(request.getBoxId(), request.getChannelId());

        // 校验时间戳
        validateTimestamp(request.getTimestamp());

        // 校验图片URL格式
        if (StringUtils.isNotBlank(request.getImageUrl())) {
            validateImageUrl(request.getImageUrl());
        }
    }

    private int calculateStateCode(StateTriple state) {
        int code = 0;
        if (state.getDoorOpen() == 1) code += 4;
        if (state.getPersonPresent() == 1) code += 2;
        if (state.getEnteringExiting() == 1) code += 1;
        // S1=1, S3=3, S5=5, S7=7, S8=8
        return switch (code) {
            case 0 -> 1;  // 000 -> S1
            case 2 -> 3;  // 010 -> S3
            case 4 -> 5;  // 100 -> S5
            case 6 -> 7;  // 110 -> S7
            case 7 -> 8;  // 111 -> S8
            default -> throw new InvalidStateException("无效的状态组合: " + code);
        };
    }

    private void validateChannelExists(String boxId, String channelId) {
        boolean exists = channelRepository.existsByBoxIdAndChannelId(boxId, channelId);
        if (!exists) {
            throw new ChannelNotFoundException(boxId, channelId);
        }
    }

    private void validateTimestamp(LocalDateTime timestamp) {
        LocalDateTime now = LocalDateTime.now();
        if (timestamp.isAfter(now.plusSeconds(5))) {
            throw new InvalidTimestampException("时间戳不能是未来时间");
        }
        if (timestamp.isBefore(now.minusMinutes(10))) {
            throw new InvalidTimestampException("时间戳过期");
        }
    }

    private void validateImageUrl(String url) {
        if (!url.startsWith("http://") && !url.startsWith("https://")) {
            throw new InvalidImageUrlException("图片URL格式错误");
        }
    }
}
```

### 5.1 幂等控制实现

#### 5.1.1 幂等键生成算法

```java
/**
 * 幂等键生成规则
 * 格式: {boxId}:{channelId}:{timestamp}
 * 示例: BOX_001:CH_001:2026-03-25T10:23:15
 *
 * 对于报警事件，追加alarmType以区分同时间的不同类型报警:
 * 格式: {boxId}:{channelId}:{timestamp}:{alarmType}
 */
public class IdempotencyKeyGenerator {

    public static String generate(String boxId, String channelId, LocalDateTime timestamp) {
        // 时间戳精确到毫秒，避免同一毫秒内的重复
        return String.format("%s:%s:%s", boxId, channelId, timestamp.toString());
    }

    public static String generateForAlarm(String boxId, String channelId,
                                          LocalDateTime timestamp, String alarmType) {
        return String.format("%s:%s:%s:%s", boxId, channelId, timestamp.toString(), alarmType);
    }
}
```

#### 5.1.2 幂等检查流程

```java
@Component
@RequiredArgsConstructor
public class IdempotencyChecker {

    private final IdempotencyRepository repository;

    // 本地缓存，减少数据库查询
    private final Cache<String, Boolean> localCache = Caffeine.newBuilder()
        .maximumSize(10000)
        .expireAfterWrite(Duration.ofMinutes(5))
        .build();

    /**
     * 检查是否为重复请求
     * 检查策略:
     * 1. 先查本地缓存 (Caffeine)
     * 2. 再查数据库 (带时间窗口)
     */
    public boolean isDuplicate(String key) {
        // 1. 本地缓存检查
        Boolean cached = localCache.getIfPresent(key);
        if (Boolean.TRUE.equals(cached)) {
            return true;
        }

        // 2. 数据库检查 (只查最近5分钟的数据)
        LocalDateTime since = LocalDateTime.now().minusMinutes(5);
        boolean exists = repository.existsByKeyAndCreatedAtAfter(key, since);

        if (exists) {
            // 加入本地缓存
            localCache.put(key, true);
        }

        return exists;
    }

    /**
     * 保存幂等记录
     * 双写策略: 本地缓存 + 数据库
     */
    @Transactional
    public void record(String key, String type) {
        // 1. 保存到数据库
        IdempotencyRecord record = new IdempotencyRecord();
        record.setKey(key);
        record.setType(type);
        record.setExpireAt(LocalDateTime.now().plusMinutes(10));
        repository.save(record);

        // 2. 更新本地缓存
        localCache.put(key, true);
    }
}
```

#### 5.1.3 幂等记录清理

```java
@Component
@RequiredArgsConstructor
public class IdempotencyCleanupTask {

    private final IdempotencyRepository repository;

    /**
     * 定时清理过期幂等记录
     * 执行频率: 每10分钟
     * 清理条件: expire_at < now()
     */
    @Scheduled(fixedRate = 10 * 60 * 1000)

    public void cleanup() {

    }
}
```

```java
@Component
@RequiredArgsConstructor
@Slf4j
public class InletEventListener {

    private final RuleEngineService ruleEngine;
    private final AlarmPushService alarmPush;
    private final DeviceMonitorService deviceMonitor;

    /**
     * 状态流事件监听
     * 异步处理，交由规则引擎进行状态机转换
     */
    @Async("inletEventExecutor")
    @EventListener
    @TransactionalEventListener(phase = TransactionPhase.AFTER_COMMIT)
    public void onStateStream(StateStreamEvent event) {
        log.debug("收到状态流事件: streamId={}, stateCode={}",
            event.getStreamId(), event.getStateCode());

        try {
            ruleEngine.processStateStream(event);
        } catch (Exception e) {
            log.error("状态流事件处理失败: streamId={}", event.getStreamId(), e);
            // 可发送告警或进入死信队列
        }
    }

    /**
     * 报警事件监听
     * 异步处理，推送前端并触发业务逻辑
     */
    @Async("inletEventExecutor")
    @EventListener
    @TransactionalEventListener(phase = TransactionPhase.AFTER_COMMIT)
    public void onAlarmEvent(AlarmEvent event) {
        log.info("收到报警事件: alarmId={}, type={}",
            event.getAlarmId(), event.getAlarmType());

        try {
            // 1. 推送实时预警到前端 (SSE)
            alarmPush.pushRealtimeAlarm(event);

            // 2. 触发规则引擎后续处理
            ruleEngine.processAlarm(event);
        } catch (Exception e) {
            log.error("报警事件处理失败: alarmId={}", event.getAlarmId(), e);
        }
    }

    /**
     * 心跳事件监听
     * 异步处理，监控盒子资源使用情况
     */
    @Async("inletEventExecutor")
    @EventListener
    @TransactionalEventListener(phase = TransactionPhase.AFTER_COMMIT)
    public void onHeartbeat(HeartbeatEvent event) {
        log.debug("收到心跳事件: boxId={}", event.getBoxCode());

        try {
            // 检查资源使用率，触发告警
            deviceMonitor.checkResourceUsage(event);
        } catch (Exception e) {
            log.error("心跳事件处理失败: boxId={}", event.getBoxCode(), e);
        }
    }
}
```
