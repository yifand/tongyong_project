## 1. 模块职责简述

系统配置模块（`pdi-system-config`）负责PDI智能监测平台的全局配置管理，包括：

- **算法开关配置**：按通道或全局控制各算法的启用/禁用状态
- **算法参数配置**：管理灵敏度、触发帧数、标准工时等算法运行参数
- **通用配置**：数据留存周期、报警声音、报表水印等系统级配置
- **业务规则配置**：状态机转换规则、PDI标准工时规则、报警阈值规则等

### 模块边界

- 向上提供REST API供前端配置界面调用
- 向下依赖PostgreSQL持久化配置数据
- 通过ConfigCache为其他模块提供高性能配置读取能力
- 配置变更时发布领域事件，通知规则引擎等模块热更新

@RestController
@RequestMapping("/api/v1/config")
@Tag(name = "系统配置管理", description = "算法配置、业务规则、通用配置管理")
@RequiredArgsConstructor
public class ConfigController {

    private final AlgorithmConfigService algorithmConfigService;
    private final BusinessRuleService businessRuleService;
    private final SystemConfigService systemConfigService;

    // ==================== 算法配置接口 ====================

    @GetMapping("/algorithm/{channelId}")
    @Operation(summary = "获取通道算法配置")
    public Result<AlgorithmConfigResponse> getAlgorithmConfig(
            @PathVariable Long channelId,
            @RequestParam(required = false) String algorithmType) {
        return Result.success(algorithmConfigService.getConfig(channelId, algorithmType));
    }

    @PutMapping("/algorithm/{channelId}")
    @Operation(summary = "更新通道算法配置")
    @PreAuthorize("hasRole('ADMIN')")
    public Result<Void> updateAlgorithmConfig(
            @PathVariable Long channelId,
            @Valid @RequestBody AlgorithmConfigRequest request) {
        algorithmConfigService.updateConfig(channelId, request);
        return Result.success();
    }

    @GetMapping("/algorithm/global")
    @Operation(summary = "获取全局算法配置")
    public Result<AlgorithmConfigResponse> getGlobalAlgorithmConfig() {
        return Result.success(algorithmConfigService.getGlobalConfig());
    }

    @PutMapping("/algorithm/global")
    @Operation(summary = "更新全局算法配置")
    @PreAuthorize("hasRole('SUPER_ADMIN')")
    public Result<Void> updateGlobalAlgorithmConfig(
            @Valid @RequestBody AlgorithmConfigRequest request) {
        algorithmConfigService.updateGlobalConfig(request);
        return Result.success();
    }

    // ==================== 业务规则接口 ====================

    @GetMapping("/rules")
    @Operation(summary = "业务规则列表")
    public Result<PageResult<BusinessRuleResponse>> listRules(
            @RequestParam(defaultValue = "1") int page,
            @RequestParam(defaultValue = "20") int size,
            @RequestParam(required = false) String ruleType) {
        return Result.success(businessRuleService.listRules(page, size, ruleType));
    }

    @GetMapping("/rules/{id}")
    @Operation(summary = "业务规则详情")
    public Result<BusinessRuleResponse> getRule(@PathVariable Long id) {
        return Result.success(businessRuleService.getRule(id));
    }

    @PutMapping("/rules/{id}")
    @Operation(summary = "更新业务规则")
    @PreAuthorize("hasRole('ADMIN')")
    public Result<Void> updateRule(
            @PathVariable Long id,
            @Valid @RequestBody BusinessRuleRequest request) {
        businessRuleService.updateRule(id, request);
        return Result.success();
    }

    @PostMapping("/rules/{id}/enable")
    @Operation(summary = "启用业务规则")
    @PreAuthorize("hasRole('ADMIN')")
    public Result<Void> enableRule(@PathVariable Long id) {
        businessRuleService.enableRule(id, true);
        return Result.success();
    }

    @PostMapping("/rules/{id}/disable")
    @Operation(summary = "禁用业务规则")
    @PreAuthorize("hasRole('ADMIN')")
    public Result<Void> disableRule(@PathVariable Long id) {
        businessRuleService.enableRule(id, false);
        return Result.success();
    }

    // ==================== 通用配置接口 ====================

    @GetMapping("/general")
    @Operation(summary = "获取通用配置")
    public Result<ConfigGroupResponse> getGeneralConfig(
            @RequestParam(required = false) String configGroup) {
        return Result.success(systemConfigService.getGeneralConfig(configGroup));
    }

    @PutMapping("/general")
    @Operation(summary = "更新通用配置")

### 2.6 ConfigCache

```java
@Component
@Slf4j
public class ConfigCache {

    private static final long CACHE_TTL_MINUTES = 10;

    // 算法开关缓存: key = algorithm:switch:{channelId}:{algorithmType}
    private final Cache<String, Boolean> algorithmSwitchCache = Caffeine.newBuilder()
            .expireAfterWrite(CACHE_TTL_MINUTES, TimeUnit.MINUTES)
            .maximumSize(1000)
            .recordStats()
            .build();

    // 算法参数缓存: key = algorithm:config:{channelId}
    private final Cache<String, AlgorithmConfigDTO> algorithmConfigCache = Caffeine.newBuilder()
            .expireAfterWrite(CACHE_TTL_MINUTES, TimeUnit.MINUTES)
            .maximumSize(500)
            .build();

    // 通用配置缓存: key = config:{configKey}
    private final Cache<String, String> generalConfigCache = Caffeine.newBuilder()
            .expireAfterWrite(CACHE_TTL_MINUTES, TimeUnit.MINUTES)
            .maximumSize(200)
            .build();

    // 业务规则缓存: key = rule:{ruleCode}
    private final Cache<String, BusinessRuleDTO> businessRuleCache = Caffeine.newBuilder()
            .expireAfterWrite(CACHE_TTL_MINUTES, TimeUnit.MINUTES)
            .maximumSize(100)
            .build();

    /**
     * 获取算法开关
     */
    public Boolean getAlgorithmSwitch(Long channelId, String algorithmType) {
        String key = buildAlgorithmSwitchKey(channelId, algorithmType);
        return algorithmSwitchCache.getIfPresent(key);
    }

    /**
     * 缓存算法开关
     */
    public void putAlgorithmSwitch(Long channelId, String algorithmType, Boolean enabled) {
        String key = buildAlgorithmSwitchKey(channelId, algorithmType);
        algorithmSwitchCache.put(key, enabled);
    }

    /**
     * 获取算法配置
     */
    public AlgorithmConfigDTO getAlgorithmConfig(Long channelId) {
        return algorithmConfigCache.getIfPresent(buildAlgorithmConfigKey(channelId));
    }

    /**
     * 缓存算法配置
     */
    public void putAlgorithmConfig(Long channelId, AlgorithmConfigDTO config) {
        algorithmConfigCache.put(buildAlgorithmConfigKey(channelId), config);
    }

    /**
     * 获取通用配置
     */
    public String getGeneralConfig(String configKey) {
        return generalConfigCache.getIfPresent(buildGeneralConfigKey(configKey));
    }

    /**
     * 缓存通用配置
     */
    public void putGeneralConfig(String configKey, String value) {
        generalConfigCache.put(buildGeneralConfigKey(configKey), value);
    }

    /**
     * 获取业务规则
     */
    public BusinessRuleDTO getBusinessRule(String ruleCode) {
        return businessRuleCache.getIfPresent(buildBusinessRuleKey(ruleCode));
    }

    /**
     * 缓存业务规则
     */
    public void putBusinessRule(String ruleCode, BusinessRuleDTO rule) {
        businessRuleCache.put(buildBusinessRuleKey(ruleCode), rule);
    }

    /**
     * 清除算法配置缓存
     */
    public void invalidateAlgorithmConfig(Long channelId) {
        algorithmConfigCache.invalidate(buildAlgorithmConfigKey(channelId));
        // 同时清除该通道的所有算法开关缓存
        algorithmSwitchCache.asMap().keySet().stream()
                .filter(key -> key.startsWith("algorithm:switch:" + channelId + ":"))
                .forEach(algorithmSwitchCache::invalidate);
        log.info("已清除通道[{}]的算法配置缓存", channelId);
    }

    /**
     * 清除通用配置缓存
     */
    public void invalidateGeneralConfig(String configKey) {
        generalConfigCache.invalidate(buildGeneralConfigKey(configKey));
    }

    /**
     * 清除业务规则缓存
     */
    public void invalidateBusinessRule(String ruleCode) {
        businessRuleCache.invalidate(buildBusinessRuleKey(ruleCode));
    }

    /**
     * 清除所有缓存
     */
    public void invalidateAll() {
        algorithmSwitchCache.invalidateAll();
        algorithmConfigCache.invalidateAll();
        generalConfigCache.invalidateAll();
        businessRuleCache.invalidateAll();
        log.info("已清除所有配置缓存");
    }

    // 缓存键构建方法
    private String buildAlgorithmSwitchKey(Long channelId, String algorithmType) {
        return String.format("algorithm:switch:%d:%s", channelId, algorithmType);
    }

    private String buildAlgorithmConfigKey(Long channelId) {
        return String.format("algorithm:config:%d", channelId);
    }

    private String buildGeneralConfigKey(String configKey) {
        return String.format("config:%s", configKey);
    }

    private String buildBusinessRuleKey(String ruleCode) {
        return String.format("rule:%s", ruleCode);
    }
}
```

public class AlgorithmConfigRequest {

    @Schema(description = "算法类型: SMOKE-抽烟检测, PDI_LEFT_FRONT-左前门, PDI_LEFT_REAR-左后门, PDI_SLIDE-滑移门")

### 5.3 缓存一致性保证

```java
@Service
@RequiredArgsConstructor
public class AlgorithmConfigServiceImpl implements AlgorithmConfigService {

    private final AlgorithmConfigRepository repository;
    private final ConfigCache configCache;
    private final ApplicationEventPublisher eventPublisher;
    private final OperationLogService operationLogService;

    @Override
    @Transactional
    public void updateConfig(Long channelId, AlgorithmConfigRequest request) {
        // 1. 查询现有配置
        AlgorithmConfig config = repository
                .findByChannelIdAndAlgorithmType(channelId, request.getAlgorithmType())
                .orElse(new AlgorithmConfig());

        // 2. 更新配置
        config.setChannelId(channelId);
        config.setAlgorithmType(request.getAlgorithmType());
        config.setEnabled(request.getEnabled());
        config.setSensitivity(request.getSensitivity());
        config.setTriggerFrames(request.getTriggerFrames());
        config.setStandardDuration(request.getStandardDuration());
        config.setEnterExitWindow(request.getEnterExitWindow());
        config.setPersonDisappearTimeout(request.getPersonDisappearTimeout());
        config.setInheritGlobal(request.getInheritGlobal());
        config.setUpdatedBy(getCurrentUserId());
        config.setUpdatedAt(LocalDateTime.now());

        // 3. 保存到数据库
        repository.save(config);

        // 4. 清除缓存
        configCache.invalidateAlgorithmConfig(channelId);

        // 5. 发布配置变更事件
        eventPublisher.publishEvent(new ConfigChangedEvent(
                this,
                ConfigType.ALGORITHM,
                channelId,
                request.getAlgorithmType()
        ));

        // 6. 记录操作日志
        operationLogService.log(OperationType.CONFIG_UPDATE,
                String.format("更新通道[%d]算法[%s]配置", channelId, request.getAlgorithmType()));
    }
}
```
