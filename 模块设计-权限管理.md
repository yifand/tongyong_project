## 1. 模块职责简述

权限管理模块（`pdi-auth-management`）负责PDI智能监测平台的用户认证、授权及权限管理功能，是系统安全的核心保障模块。

### 核心职责

| 职责 | 说明 |
|------|------|
| **用户管理** | 用户的增删改查、密码策略管理、用户状态管理 |
| **角色管理** | 预设角色定义、权限粒度配置、角色分配 |
| **登录认证** | 用户名密码认证、JWT令牌签发与验证、登出处理 |
| **权限校验** | 功能权限校验（API级别）、数据权限校验（站点隔离） |
| **登录锁定** | 连续登录失败锁定机制，防止暴力破解 |

### 模块依赖

- 依赖 `pdi-common` 模块获取公共实体、工具类
- 被 `pdi-alarm-center`、`pdi-behavior-archive` 等模块依赖用于权限校验

#### AuthServiceImpl

```java
@Service
@RequiredArgsConstructor
@Slf4j
public class AuthServiceImpl implements AuthService {

    private final UserRepository userRepository;
    private final JwtTokenProvider jwtTokenProvider;
    private final PasswordEncoder passwordEncoder;
    private final LoginLockService loginLockService;
    private final TokenBlacklistService tokenBlacklistService;

    @Override
    @Transactional
    public LoginResponse login(String username, String password) {
        // 1. 检查登录锁定
        if (loginLockService.isLocked(username)) {
            long remainingSeconds = loginLockService.getRemainingLockTime(username);
            throw new BizException(ErrorCode.ACCOUNT_LOCKED,
                "账户已锁定，请" + remainingSeconds + "秒后重试");
        }

        // 2. 查询用户
        User user = userRepository.findByUsernameAndDeletedAtIsNull(username)
            .orElseThrow(() -> {
                loginLockService.recordFailedAttempt(username);
                return new BizException(ErrorCode.INVALID_CREDENTIALS, "用户名或密码错误");
            });

        // 3. 校验状态
        if (user.getStatus() != UserStatus.ENABLED.getCode()) {
            throw new BizException(ErrorCode.ACCOUNT_DISABLED, "账户已禁用");
        }

        // 4. 校验密码
        if (!passwordEncoder.matches(password, user.getPassword())) {
            loginLockService.recordFailedAttempt(username);
            throw new BizException(ErrorCode.INVALID_CREDENTIALS, "用户名或密码错误");
        }

        // 5. 登录成功，清除失败记录
        loginLockService.clearFailedAttempts(username);

        // 6. 生成Token
        String accessToken = jwtTokenProvider.generateAccessToken(user);
        String refreshToken = jwtTokenProvider.generateRefreshToken(user);

        // 7. 更新最后登录时间
        user.setLastLoginAt(LocalDateTime.now());
        userRepository.save(user);

        return LoginResponse.builder()
            .accessToken(accessToken)
            .refreshToken(refreshToken)
            .expiresIn(jwtTokenProvider.getAccessTokenExpiration())
            .user(mapToUserVO(user))
            .build();
    }

    @Override
    public void logout(String token) {
        String jwt = jwtTokenProvider.resolveToken(token);
        if (jwt != null && jwtTokenProvider.validateToken(jwt)) {
            long expiration = jwtTokenProvider.getExpirationDate(jwt).getTime() - System.currentTimeMillis();
            if (expiration > 0) {
                tokenBlacklistService.addToBlacklist(jwt, expiration);
            }
        }
    }

    @Override
    public String refreshAccessToken(String refreshToken) {
        if (!jwtTokenProvider.validateToken(refreshToken)) {
            throw new BizException(ErrorCode.INVALID_TOKEN, "刷新令牌无效或已过期");
        }

        String userId = jwtTokenProvider.getUserId(refreshToken);
        User user = userRepository.findById(Long.valueOf(userId))
            .orElseThrow(() -> new BizException(ErrorCode.USER_NOT_FOUND, "用户不存在"));

        return jwtTokenProvider.generateAccessToken(user);
    }
}
```

```java
@Service
@RequiredArgsConstructor
public class UserServiceImpl implements UserService {

    private final UserRepository userRepository;
    private final RoleRepository roleRepository;
    private final UserRoleRepository userRoleRepository;
    private final PasswordEncoder passwordEncoder;
    private final PasswordValidator passwordValidator;

    @Override
    @DataScope(siteIdField = "siteId")
    public PageResult<UserResponse> listUsers(int page, int size, String keyword, Long siteId) {
        Pageable pageable = PageRequest.of(page - 1, size, Sort.by("createdAt").descending());

        Specification<User> spec = (root, query, cb) -> {
            List<Predicate> predicates = new ArrayList<>();
            predicates.add(cb.isNull(root.get("deletedAt")));

            if (StringUtils.hasText(keyword)) {
                predicates.add(cb.or(
                    cb.like(root.get("username"), "%" + keyword + "%"),
                    cb.like(root.get("name"), "%" + keyword + "%")
                ));
            }

            if (siteId != null) {
                predicates.add(cb.equal(root.get("siteId"), siteId));
            }

            return cb.and(predicates.toArray(new Predicate[0]));
        };

        Page<User> userPage = userRepository.findAll(spec, pageable);
        return PageResult.of(userPage.map(this::mapToResponse));
    }

    @Override
    @Transactional
    public Long createUser(CreateUserRequest request) {
        // 校验密码策略
        passwordValidator.validate(request.getPassword());

        // 检查用户名唯一性
        if (userRepository.existsByUsername(request.getUsername())) {
            throw new BizException(ErrorCode.USER_EXISTS, "用户名已存在");
        }

        User user = new User();
        user.setUsername(request.getUsername());
        user.setPassword(passwordEncoder.encode(request.getPassword()));
        user.setName(request.getName());
        user.setPhone(request.getPhone());
        user.setSiteId(request.getSiteId());
        user.setStatus(UserStatus.ENABLED.getCode());

        userRepository.save(user);

        // 分配角色
        if (CollectionUtils.isNotEmpty(request.getRoleIds())) {
            assignRoles(user.getId(), request.getRoleIds());
        }

        return user.getId();
    }

    @Override
    @Transactional
    public void resetPassword(Long id, String newPassword) {
        passwordValidator.validate(newPassword);

        User user = userRepository.findById(id)
            .orElseThrow(() -> new BizException(ErrorCode.USER_NOT_FOUND, "用户不存在"));

        user.setPassword(passwordEncoder.encode(newPassword));
        userRepository.save(user);
    }

    // ... 其他方法实现
}
```

### 2.4 Security 组件

#### JwtTokenProvider

```java
@Component
@Slf4j
public class JwtTokenProvider {

    @Value("${jwt.secret}")
    private String jwtSecret;

    @Value("${jwt.access-token-expiration:7200000}") // 2小时
    private long accessTokenExpiration;

    @Value("${jwt.refresh-token-expiration:604800000}") // 7天
    private long refreshTokenExpiration;

    private SecretKey key;

    @PostConstruct
    public void init() {
        this.key = Keys.hmacShaKeyFor(jwtSecret.getBytes(StandardCharsets.UTF_8));
    }

    /**
     * 生成Access Token
     */
    public String generateAccessToken(User user) {
        Date now = new Date();
        Date expiryDate = new Date(now.getTime() + accessTokenExpiration);

        return Jwts.builder()
            .subject(String.valueOf(user.getId()))
            .claim("username", user.getUsername())
            .claim("name", user.getName())
            .claim("role", user.getPrimaryRole())
            .claim("siteId", user.getSiteId())
            .claim("type", "ACCESS")
            .issuedAt(now)
            .expiration(expiryDate)
            .signWith(key, Jwts.SIG.HS256)
            .compact();
    }

    /**
     * 生成Refresh Token
     */
    public String generateRefreshToken(User user) {
        Date now = new Date();
        Date expiryDate = new Date(now.getTime() + refreshTokenExpiration);

        return Jwts.builder()
            .subject(String.valueOf(user.getId()))
            .claim("type", "REFRESH")
            .issuedAt(now)
            .expiration(expiryDate)
            .signWith(key, Jwts.SIG.HS256)
            .compact();
    }

    /**
     * 验证Token有效性
     */
    public boolean validateToken(String token) {
        try {
            Jwts.parser().verifyWith(key).build().parseSignedClaims(token);
            return true;
        } catch (JwtException | IllegalArgumentException e) {
            log.warn("Invalid JWT token: {}", e.getMessage());
            return false;
        }
    }

    /**
     * 从Token获取用户ID
     */
    public String getUserId(String token) {
        Claims claims = parseClaims(token);
        return claims.getSubject();
    }

    /**
     * 从Token获取用户名
     */
    public String getUsername(String token) {
        Claims claims = parseClaims(token);
        return claims.get("username", String.class);
    }

    /**
     * 从Token获取角色
     */
    public String getRole(String token) {
        Claims claims = parseClaims(token);
        return claims.get("role", String.class);
    }

    /**
     * 从Token获取站点ID
     */
    public Long getSiteId(String token) {
        Claims claims = parseClaims(token);
        Integer siteId = claims.get("siteId", Integer.class);
        return siteId != null ? siteId.longValue() : null;
    }

    /**
     * 解析Claims
     */
    public Claims parseClaims(String token) {
        return Jwts.parser()
            .verifyWith(key)
            .build()
            .parseSignedClaims(token)
            .getPayload();
    }

    /**
     * 获取Token过期时间
     */
    public Date getExpirationDate(String token) {
        return parseClaims(token).getExpiration();
    }

    /**
     * 从请求头解析Token
     */
    public String resolveToken(String bearerToken) {
        if (StringUtils.hasText(bearerToken) && bearerToken.startsWith("Bearer ")) {
            return bearerToken.substring(7);
        }
        return null;
    }

    public long getAccessTokenExpiration() {
        return accessTokenExpiration / 1000; // 返回秒
    }
}
```

#### JwtAuthenticationFilter

```java
@Component
@RequiredArgsConstructor
public class JwtAuthenticationFilter extends OncePerRequestFilter {

    private final JwtTokenProvider jwtTokenProvider;
    private final TokenBlacklistService tokenBlacklistService;
    private final UserDetailsServiceImpl userDetailsService;

    @Override
    protected void doFilterInternal(HttpServletRequest request,
                                      HttpServletResponse response,
                                      FilterChain filterChain) throws ServletException, IOException {

        String token = jwtTokenProvider.resolveToken(request.getHeader("Authorization"));

        if (token != null && jwtTokenProvider.validateToken(token)) {
            // 检查Token是否在黑名单中
            if (tokenBlacklistService.isBlacklisted(token)) {
                filterChain.doFilter(request, response);
                return;
            }

            String username = jwtTokenProvider.getUsername(token);
            UserDetails userDetails = userDetailsService.loadUserByUsername(username);

            UsernamePasswordAuthenticationToken authentication =
                new UsernamePasswordAuthenticationToken(
                    userDetails, null, userDetails.getAuthorities());
            authentication.setDetails(new WebAuthenticationDetailsSource().buildDetails(request));

            SecurityContextHolder.getContext().setAuthentication(authentication);
        }

        filterChain.doFilter(request, response);
    }
}
```

```java
@Service
@RequiredArgsConstructor
public class UserDetailsServiceImpl implements UserDetailsService {

    private final UserRepository userRepository;
    private final RoleRepository roleRepository;

    @Override
    public UserDetails loadUserByUsername(String username) throws UsernameNotFoundException {
        User user = userRepository.findByUsernameAndDeletedAtIsNull(username)
            .orElseThrow(() -> new UsernameNotFoundException("用户不存在: " + username));

        List<Role> roles = roleRepository.findByUserId(user.getId());

        List<GrantedAuthority> authorities = roles.stream()
            .flatMap(role -> {
                List<GrantedAuthority> auths = new ArrayList<>();
                auths.add(new SimpleGrantedAuthority("ROLE_" + role.getCode()));
                // 添加权限列表
                if (StringUtils.hasText(role.getPermissions())) {
                    List<String> perms = JsonUtils.parseList(role.getPermissions(), String.class);
                    perms.forEach(perm -> auths.add(new SimpleGrantedAuthority(perm)));
                }
                return auths.stream();
            })
            .collect(Collectors.toList());

        return new JwtUserDetails(
            user.getId(),
            user.getUsername(),
            user.getPassword(),
            user.getSiteId(),
            user.getStatus() == UserStatus.ENABLED.getCode(),
            authorities
        );
    }
}
```

### 2.5 登录锁定服务

#### LoginLockService

```java
@Service
public class LoginLockService {

    private static final int MAX_FAILED_ATTEMPTS = 5;
    private static final long LOCK_DURATION_MINUTES = 5;

    private final Cache<String, Integer> failedAttemptsCache;
    private final Cache<String, Long> lockCache;

    public LoginLockService() {
        this.failedAttemptsCache = Caffeine.newBuilder()
            .maximumSize(1000)
            .expireAfterWrite(Duration.ofMinutes(30))
            .build();

        this.lockCache = Caffeine.newBuilder()
            .maximumSize(1000)
            .expireAfterWrite(Duration.ofMinutes(LOCK_DURATION_MINUTES))
            .build();
    }

    /**
     * 记录登录失败尝试
     */
    public void recordFailedAttempt(String username) {
        Integer attempts = failedAttemptsCache.get(username, k -> 0);
        attempts = attempts + 1;
        failedAttemptsCache.put(username, attempts);

        if (attempts >= MAX_FAILED_ATTEMPTS) {
            lock(username);
        }
    }

    /**
     * 锁定账户
     */
    public void lock(String username) {
        lockCache.put(username, System.currentTimeMillis());
        failedAttemptsCache.invalidate(username);
    }

    /**
     * 检查账户是否被锁定
     */
    public boolean isLocked(String username) {
        return lockCache.getIfPresent(username) != null;
    }

    /**
     * 获取剩余锁定时间（秒）
     */
    public long getRemainingLockTime(String username) {
        Long lockTime = lockCache.getIfPresent(username);
        if (lockTime == null) {
            return 0;
        }

        long elapsed = System.currentTimeMillis() - lockTime;
        long remaining = (LOCK_DURATION_MINUTES * 60 * 1000) - elapsed;
        return Math.max(0, remaining / 1000);
    }

    /**
     * 清除失败记录（登录成功时调用）
     */
    public void clearFailedAttempts(String username) {
        failedAttemptsCache.invalidate(username);
        lockCache.invalidate(username);
    }
}
```

### 2.6 密码策略验证器

#### PasswordValidator

```java
@Component
public class PasswordValidator {

    @Value("${password.min-length:8}")
    private int minLength;

    @Value("${password.require-uppercase:true}")
    private boolean requireUppercase;

    @Value("${password.require-lowercase:true}")
    private boolean requireLowercase;

    @Value("${password.require-digit:true}")
    private boolean requireDigit;

    /**
     * 验证密码是否符合策略
     */
    public void validate(String password) {
        if (!StringUtils.hasText(password)) {
            throw new BizException(ErrorCode.INVALID_PASSWORD, "密码不能为空");
        }

        if (password.length() < minLength) {
            throw new BizException(ErrorCode.INVALID_PASSWORD,
                "密码长度不能少于" + minLength + "位");
        }

        if (requireUppercase && !containsUppercase(password)) {
            throw new BizException(ErrorCode.INVALID_PASSWORD,
                "密码必须包含大写字母");
        }

        if (requireLowercase && !containsLowercase(password)) {
            throw new BizException(ErrorCode.INVALID_PASSWORD,
                "密码必须包含小写字母");
        }

        if (requireDigit && !containsDigit(password)) {
            throw new BizException(ErrorCode.INVALID_PASSWORD,
                "密码必须包含数字");
        }
    }

    private boolean containsUppercase(String password) {
        return password.chars().anyMatch(Character::isUpperCase);
    }

    private boolean containsLowercase(String password) {
        return password.chars().anyMatch(Character::isLowerCase);
    }

    private boolean containsDigit(String password) {
        return password.chars().anyMatch(Character::isDigit);
    }
}
```

```java
@RequiredArgsConstructor
@Slf4j
public class DataScopeAspect {

    private static final String SUPER_ADMIN_ROLE = "SUPER_ADMIN";

    /**
     * 拦截带有@DataScope注解的方法
     */
    @Before("@annotation(dataScope)")
    public void before(JoinPoint point, DataScope dataScope) {
        if (dataScope.ignore()) {
            return;
        }

        Authentication authentication = SecurityContextHolder.getContext().getAuthentication();
        if (authentication == null || !(authentication.getPrincipal() instanceof JwtUserDetails)) {
            return;
        }

        JwtUserDetails userDetails = (JwtUserDetails) authentication.getPrincipal();

        // 超级管理员跳过数据权限过滤
        if (isSuperAdmin(userDetails)) {
            log.debug("Super admin access, skip data scope filter");
            return;
        }

        // 将当前用户的siteId存入ThreadLocal，供Repository层使用
        Long siteId = userDetails.getSiteId();
        DataScopeContext.setCurrentSiteId(siteId);
    }

    @After("@annotation(com.vdc.pdi.auth.aspect.DataScope)")
    public void after() {
        DataScopeContext.clear();
    }

    private boolean isSuperAdmin(JwtUserDetails userDetails) {
        return userDetails.getAuthorities().stream()
            .anyMatch(auth -> auth.getAuthority().equals("ROLE_" + SUPER_ADMIN_ROLE));
    }
}
```

```java
class LoginLockServiceTest {

    private LoginLockService loginLockService;

    @BeforeEach
    void setUp() {
        loginLockService = new LoginLockService();
    }

    @Test
    @DisplayName("记录失败尝试：5次后锁定账户")
    void recordFailedAttempt_LockAfter5Attempts() {
        // Given
        String username = "testuser";

        // When - 记录4次失败
        for (int i = 0; i < 4; i++) {
            loginLockService.recordFailedAttempt(username);
        }

        // Then - 未锁定
        assertThat(loginLockService.isLocked(username)).isFalse();

        // When - 第5次失败
        loginLockService.recordFailedAttempt(username);

        // Then - 已锁定
        assertThat(loginLockService.isLocked(username)).isTrue();
    }

    @Test
    @DisplayName("登录成功后清除失败记录")
    void clearFailedAttempts_AfterSuccess() {
        // Given
        String username = "testuser";
        loginLockService.recordFailedAttempt(username);
        loginLockService.recordFailedAttempt(username);

        // When
        loginLockService.clearFailedAttempts(username);

        // Then
        assertThat(loginLockService.isLocked(username)).isFalse();
    }

    @Test
    @DisplayName("获取剩余锁定时间")
    void getRemainingLockTime() {
        // Given
        String username = "testuser";
        for (int i = 0; i < 5; i++) {
            loginLockService.recordFailedAttempt(username);
        }

        // When
        long remaining = loginLockService.getRemainingLockTime(username);

        // Then
        assertThat(remaining).isGreaterThan(0).isLessThanOrEqualTo(300); // 5分钟=300秒
    }
}
```
